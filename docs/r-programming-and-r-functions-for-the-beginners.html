# further usage of foor loop
a <- c("a1","a2","a3","a4","a5","a6")
b <- c("b1","b2","b3","b4","b5","b6")
c <- c("c1","c2","c3","c4","c5","c6")

df <- data.frame(a,b,c,stringsAsFactors = FALSE)

for(colVlaue in 1:ncol(df)){
  for(rowValue in 1:nrow(df)){
    print(df[rowValue,colVlaue])
  }
}

for(colValue in 1:ncol(df)){
  for(rowValue in 1:nrow(df)){
    if(df[rowValue,colValue] == "c6"){
      print(c(rowValue, colValue))
    }
  }
}
